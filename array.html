C Code

void insert(int arr[], int *size, int element, int index) {
    for (int i = *size; i > index; i--)
        arr[i] = arr[i - 1];
    arr[index] = element;
    (*size)++;
}

void delete(int arr[], int *size, int index) {
    for (int i = index; i < *size - 1; i++)
        arr[i] = arr[i + 1];
    (*size)--;
}
